```python
    def subscribe(self, user_id, subscription_type, category, frequency='daily'):
        """订阅"""
        self.db.execute("""
            INSERT OR REPLACE INTO user_subscriptions 
            (user_id, subscription_type, category, frequency, is_active)
            VALUES (?, ?, ?, ?, TRUE)
        """, (user_id, subscription_type, category, frequency))
        self.db.commit()
    
    def unsubscribe(self, user_id, subscription_type, category):
        """取消订阅"""
        self.db.execute("""
            UPDATE user_subscriptions 
            SET is_active = FALSE
            WHERE user_id = ? AND subscription_type = ? AND category = ?
        """, (user_id, subscription_type, category))
        self.db.commit()
    
    def send_daily_digest(self):
        """发送每日精选"""
        # 获取订阅用户
        subscribers = self.db.execute("""
            SELECT u.id, u.email, u.username
            FROM users u
            JOIN user_subscriptions s ON u.id = s.user_id
            WHERE s.subscription_type = 'email'
            AND s.category = 'daily_digest'
            AND s.frequency = 'daily'
            AND s.is_active = TRUE
        """).fetchall()
        
        for user_id, email, username in subscribers:
            # 获取今日内容
            content = self.get_daily_content(user_id)
            
            # 渲染邮件
            html = self.templates['daily_digest'].render(
                username=username,
                warning_letters=content['warning_letters'],
                regulations=content['regulations'],
                industry_news=content['industry_news'],
                unsubscribe_url=f"https://fda-monitor.com/unsubscribe?user_id={user_id}"
            )
            
            # 发送邮件
            self.send_email(email, 'FDA 警告信平台 - 每日精选', html)
    
    def get_daily_content(self, user_id):
        """获取每日内容"""
        # 获取用户偏好
        preferences = self.db.execute("""
            SELECT category FROM user_subscriptions
            WHERE user_id = ? AND is_active = TRUE
        """, (user_id,)).fetchall()
        
        categories = [p[0] for p in preferences]
        
        content = {
            'warning_letters': [],
            'regulations': [],
            'industry_news': []
        }
        
        # 获取最新警告信
        if 'fda_warnings' in categories or 'daily_digest' in categories:
            content['warning_letters'] = self.db.execute("""
                SELECT company_name, subject, posted_date
                FROM fda_warning_letters
                WHERE posted_date >= date('now', '-1 day')
                ORDER BY posted_date DESC
                LIMIT 5
            """).fetchall()
        
        # 获取法规更新
        if 'regulations' in categories or 'daily_digest' in categories:
            content['regulations'] = self.db.execute("""
                SELECT title, source, publish_date
                FROM regulations
                WHERE publish_date >= date('now', '-1 day')
                ORDER BY publish_date DESC
                LIMIT 5
            """).fetchall()
        
        # 获取行业资讯
        if 'industry_news' in categories or 'daily_digest' in categories:
            content['industry_news'] = self.db.execute("""
                SELECT title, summary, publish_date
                FROM news
                WHERE publish_date >= date('now', '-1 day')
                ORDER BY publish_date DESC
                LIMIT 5
            """).fetchall()
        
        return content
    
    def send_email(self, to_email, subject, html_content):
        """发送邮件"""
        msg = MIMEMultipart('alternative')
        msg['From'] = 'noreply@fda-monitor.com'
        msg['To'] = to_email
        msg['Subject'] = subject
        
        msg.attach(MIMEText(html_content, 'html'))
        
        with smtplib.SMTP('smtp.gmail.com', 587) as server:
            server.starttls()
            server.login('your-email@gmail.com', 'your-password')
            server.send_message(msg)
```

## 四、高级搜索功能

### 4.1 搜索增强

```python
# advanced_search.py
import sqlite3
from whoosh.index import open_dir
from whoosh.qparser import MultifieldParser, OrGroup
from whoosh.highlight import ContextFragmenter
from datetime import datetime, timedelta

class AdvancedSearch:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/fda_warning.db')
        self.index_dir = '/root/data/search_index'
        self.ix = open_dir(self.index_dir)
    
    def search(
        self,
        query: str,
        category: str = None,
        date_from: str = None,
        date_to: str = None,
        risk_level: str = None,
        violation_type: str = None,
        country: str = None,
        page: int = 1,
        page_size: int = 20
    ):
        """高级搜索"""
        with self.ix.searcher() as searcher:
            # 构建查询
            query_parts = []
            
            # 文本查询
            if query:
                query_parts.append(query)
            
            # 分类筛选
            if category:
                query_parts.append(f"category:{category}")
            
            # 风险等级筛选
            if risk_level:
                query_parts.append(f"risk_level:{risk_level}")
            
            # 违规类型筛选
            if violation_type:
                query_parts.append(f"violation_type:{violation_type}")
            
            # 国家筛选
            if country:
                query_parts.append(f"country:{country}")
            
            # 组合查询
            final_query = " AND ".join(query_parts) if query_parts else "*"
            
            parser = MultifieldParser(
                ["title", "content", "tags", "company_name"],
                self.ix.schema,
                group=OrGroup
            )
            q = parser.parse(final_query)
            
            # 执行搜索
            results = searcher.search(q, limit=page * page_size)
            
            # 日期筛选
            if date_from or date_to:
                filtered_results = []
                for hit in results:
                    hit_date = hit.get('date')
                    if hit_date:
                        if date_from and hit_date < date_from:
                            continue
                        if date_to and hit_date > date_to:
                            continue
                    filtered_results.append(hit)
                results = filtered_results
            
            # 分页
            start = (page - 1) * page_size
            end = start + page_size
            page_results = results[start:end]
            
            # 高亮显示
            fragmenter = ContextFragmenter(maxchars=200, surround=80)
            
            return {
                "total": len(results),
                "items": [
                    {
                        "id": hit['id'],
                        "title": hit['title'],
                        "content": hit.get('content', ''),
                        "category": hit.get('category', ''),
                        "tags": hit.get('tags', ''),
                        "date": hit.get('date', ''),
                        "highlight": hit.highlights("content", fragmenter=fragmenter)
                    }
                    for hit in page_results
                ],
                "page": page,
                "page_size": page_size,
                "filters": {
                    "category": category,
                    "date_from": date_from,
                    "date_to": date_to,
                    "risk_level": risk_level,
                    "violation_type": violation_type,
                    "country": country
                }
            }
    
    def get_suggestions(self, query: str, limit: int = 5):
        """获取搜索建议"""
        suggestions = []
        
        # 公司名称建议
        company_suggestions = self.db.execute("""
            SELECT DISTINCT company_name
            FROM fda_warning_letters
            WHERE company_name LIKE ?
            LIMIT ?
        """, (f'%{query}%', limit)).fetchall()
        
        suggestions.extend([
            {"type": "company", "text": s[0]}
            for s in company_suggestions
        ])
        
        # 违规类型建议
        violation_suggestions = self.db.execute("""
            SELECT DISTINCT violation_type
            FROM fda_analysis
            WHERE violation_type LIKE ?
            LIMIT ?
        """, (f'%{query}%', limit)).fetchall()
        
        suggestions.extend([
            {"type": "violation", "text": s[0]}
            for s in violation_suggestions
        ])
        
        return suggestions[:limit]
    
    def get_popular_searches(self, limit: int = 10):
        """获取热门搜索"""
        # 从搜索日志中获取热门搜索
        popular = self.db.execute("""
            SELECT query, COUNT(*) as count
            FROM search_logs
            WHERE created_at >= date('now', '-7 days')
            GROUP BY query
            ORDER BY count DESC
            LIMIT ?
        """, (limit,)).fetchall()
        
        return [{"query": p[0], "count": p[1]} for p in popular]
```

```python
from fastapi import FastAPI, Query
from typing import Optional, List
from datetime import date

app = FastAPI()

    q: str = Query("", min_length=0),
    category: Optional[str] = None,
    date_from: Optional[date] = None,
    date_to: Optional[date] = None,
    risk_level: Optional[str] = None,
    violation_type: Optional[str] = None,
    country: Optional[str] = None,
    sort_by: str = Query("relevance", regex="^(relevance|date|company)$"),
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100)
):
    """高级搜索API"""
    from advanced_search import AdvancedSearch
    
    searcher = AdvancedSearch()
    
    results = searcher.search(
        query=q,
        category=category,
        date_from=date_from.isoformat() if date_from else None,
        date_to=date_to.isoformat() if date_to else None,
        risk_level=risk_level,
        violation_type=violation_type,
        country=country,
        page=page,
        page_size=page_size
    )
```

### 5.1 推荐算法

```python
# recommendation.py
import sqlite3
from collections import Counter
from datetime import datetime, timedelta

class RecommendationService:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/users.db')
        self.fda_db = sqlite3.connect('/root/data/fda_warning.db')
    
    def get_personalized_recommendations(self, user_id: int, limit: int = 10):
        """获取个性化推荐"""
        # 获取用户历史行为
        history = self.get_user_history(user_id)
        
        # 获取用户偏好
        preferences = self.get_user_preferences(user_id)
        
        # 基于内容的推荐
        content_based = self.content_based_recommendations(history, preferences, limit)
        
        # 基于协同过滤的推荐
        collaborative = self.collaborative_recommendations(user_id, limit)
        
        # 合并推荐结果
        recommendations = self.merge_recommendations(content_based, collaborative, limit)
        
        return recommendations
    
    def get_user_history(self, user_id: int):
        """获取用户历史"""
        # 浏览历史
        views = self.db.execute("""
            SELECT item_type, item_id, viewed_at
            FROM user_views
            WHERE user_id = ?
            ORDER BY viewed_at DESC
            LIMIT 100
        """, (user_id,)).fetchall()
        
        # 收藏历史
        favorites = self.db.execute("""
            SELECT item_type, item_id, created_at
            FROM user_favorites
            WHERE user_id = ?
            ORDER BY created_at DESC
        """, (user_id,)).fetchall()
        
        # 搜索历史
        searches = self.db.execute("""
            SELECT query, searched_at
            FROM user_searches
            WHERE user_id = ?
            ORDER BY searched_at DESC
            LIMIT 50
        """, (user_id,)).fetchall()
        
        return {
            "views": views,
            "favorites": favorites,
            "searches": searches
        }
    
    def get_user_preferences(self, user_id: int):
        """获取用户偏好"""
        # 从历史中提取偏好
        history = self.get_user_history(user_id)
        
        # 分析浏览的警告信
        warning_letters = [v for v in history["views"] if v[0] == "warning_letter"]
        
        # 分析收藏的内容
        favorites = [f for f in history["favorites"]]
        
        # 提取关键词
        keywords = []
        for view in warning_letters:
            letter = self.fda_db.execute("""
                SELECT company_name, violation_type, issuing_office
                FROM fda_warning_letters l
                LEFT JOIN fda_analysis a ON l.id = a.letter_id
                WHERE l.id = ?
            """, (view[1],)).fetchone()
            
            if letter:
                keywords.extend(letter)
        
        # 统计关键词频率
        keyword_counts = Counter(keywords)
        
        return {
            "top_keywords": keyword_counts.most_common(10),
            "preferred_categories": self.get_preferred_categories(favorites)
        }
    
    def get_preferred_categories(self, favorites):
        """获取偏好类别"""
        categories = []
        for fav in favorites:
            if fav[0] == "warning_letter":
                categories.append("FDA警告信")
            elif fav[0] == "regulation":
                categories.append("法规信息")
            elif fav[0] == "news":
                categories.append("行业资讯")
        
        return Counter(categories).most_common(5)
    
    def content_based_recommendations(self, history, preferences, limit):
        """基于内容的推荐"""
        recommendations = []
        
        # 获取与用户偏好相似的内容
        for keyword, count in preferences["top_keywords"]:
            # 搜索相似内容
            similar_content = self.fda_db.execute("""
                SELECT l.id, l.company_name, l.subject, l.posted_date,
                       a.violation_type, a.risk_level
                FROM fda_warning_letters l
                LEFT JOIN fda_analysis a ON l.id = a.letter_id
                WHERE l.company_name LIKE ? OR a.violation_type LIKE ?
                ORDER BY l.posted_date DESC
                LIMIT 5
            """, (f'%{keyword}%', f'%{keyword}%')).fetchall()
            
            for content in similar_content:
                recommendations.append({
                    "id": content[0],
                    "type": "warning_letter",
                    "title": content[1],
                    "description": content[2],
                    "date": content[3],
                    "score": count * 0.1,
                    "reason": f"与您关注的 '{keyword}' 相关"
                })
        
        # 按分数排序
        recommendations.sort(key=lambda x: x["score"], reverse=True)
        
        return recommendations[:limit]
    
    def collaborative_recommendations(self, user_id: int, limit: int):
        """基于协同过滤的推荐"""
        # 获取相似用户
        similar_users = self.find_similar_users(user_id)
        
        recommendations = []
        
        for similar_user in similar_users:
            # 获取相似用户喜欢的内容
            user_favorites = self.db.execute("""
                SELECT item_type, item_id
                FROM user_favorites
                WHERE user_id = ?
            """, (similar_user["id"],)).fetchall()
            
            for fav in user_favorites:
                # 检查当前用户是否已经喜欢
                is_favorited = self.db.execute("""
                    SELECT COUNT(*) FROM user_favorites
                    WHERE user_id = ? AND item_type = ? AND item_id = ?
                """, (user_id, fav[0], fav[1])).fetchone()[0]
                
                if not is_favorited:
                    recommendations.append({
                        "id": fav[1],
                        "type": fav[0],
                        "score": similar_user["similarity"],
                        "reason": f"与您相似的用户也喜欢"
                    })
        
        # 按分数排序
        recommendations.sort(key=lambda x: x["score"], reverse=True)
        
        return recommendations[:limit]
    
    def find_similar_users(self, user_id: int):
        """查找相似用户"""
        # 获取当前用户的收藏
        user_favorites = self.db.execute("""
            SELECT item_type, item_id
            FROM user_favorites
            WHERE user_id = ?
        """, (user_id,)).fetchall()
        
        similar_users = []
        
        # 查找有相似收藏的用户
        for fav in user_favorites:
            users_with_same_favorite = self.db.execute("""
                SELECT user_id, COUNT(*) as common
                FROM user_favorites
                WHERE item_type = ? AND item_id = ? AND user_id != ?
                GROUP BY user_id
                ORDER BY common DESC
                LIMIT 10
            """, (fav[0], fav[1], user_id)).fetchall()
            
            for user in users_with_same_favorite:
                similarity = user[1] / len(user_favorites) if user_favorites else 0
                similar_users.append({
                    "id": user[0],
                    "similarity": similarity
                })
        
        # 按相似度排序
        similar_users.sort(key=lambda x: x["similarity"], reverse=True)
        
        return similar_users[:5]
    
    def merge_recommendations(self, content_based, collaborative, limit):
        """合并推荐结果"""
        # 合并两个列表
        all_recommendations = content_based + collaborative
        
        # 去重
        seen = set()
        unique_recommendations = []
        
        for rec in all_recommendations:
            key = (rec["type"], rec["id"])
            if key not in seen:
                seen.add(key)
                unique_recommendations.append(rec)
        
        # 按分数排序
        unique_recommendations.sort(key=lambda x: x.get("score", 0), reverse=True)
        
        return unique_recommendations[:limit]
    
    def record_view(self, user_id: int, item_type: str, item_id: int):
        """记录浏览"""
        self.db.execute("""
            INSERT INTO user_views (user_id, item_type, item_id, viewed_at)
            VALUES (?, ?, ?, CURRENT_TIMESTAMP)
        """, (user_id, item_type, item_id))
        self.db.commit()
    
    def record_search(self, user_id: int, query: str):
        """记录搜索"""
        self.db.execute("""
            INSERT INTO user_searches (user_id, query, searched_at)
            VALUES (?, ?, CURRENT_TIMESTAMP)
        """, (user_id, query))
        self.db.commit()
```
