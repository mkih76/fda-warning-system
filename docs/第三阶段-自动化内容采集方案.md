```python
# fda_crawler.py
import scrapy
from datetime import datetime
import sqlite3
import requests
import time
import random

class FDALetterSpider(scrapy.Spider):
    name = 'fda_letters'
    allowed_domains = ['fda.gov']
    start_urls = ['https://www.fda.gov/inspections-compliance-enforcement-and-criminal-investigations/warning-letters']
    
    def __init__(self):
        self.db = sqlite3.connect('/root/data/fda_warning.db')
        self.create_tables()
    
    def create_tables(self):
        """创建数据库表"""
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS fda_warning_letters (
                id INTEGER PRIMARY KEY,
                fda_id TEXT UNIQUE NOT NULL,
                company_name TEXT NOT NULL,
                subject TEXT,
                issuing_office TEXT,
                issue_date DATE,
                posted_date DATE,
                status TEXT DEFAULT 'active',
                url TEXT,
                full_text TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.db.commit()
    
    def parse(self, response):
        """解析警告信列表页"""
        # 获取警告信链接
        letters = response.css('table.views-table tbody tr')
        
        for letter in letters:
            link = letter.css('td.views-field views-field-title a::attr(href)').get()
            if link:
                yield response.follow(link, self.parse_letter)
        
        # 翻页
        next_page = response.css('li.pager__item--next a::attr(href)').get()
        if next_page:
            yield response.follow(next_page, self.parse)
    
    def parse_letter(self, response):
        """解析警告信详情页"""
        # 提取基本信息
        fda_id = response.url.split('/')[-1]
        company_name = response.css('h1::text').get('').strip()
        subject = response.css('h1::text').get('').strip()
        
        # 提取发布信息
        posted_date = response.css('time::text').get('')
        issuing_office = response.css('field--name-field-issuing-office::text').get('')
        
        # 提取全文
        full_text = response.css('article::text').getall()
        full_text = '\n'.join(full_text).strip()
        
        # 保存到数据库
        self.db.execute("""
            INSERT OR REPLACE INTO fda_warning_letters 
            (fda_id, company_name, subject, issuing_office, posted_date, url, full_text)
            VALUES (?, ?, ?, ?, ?, ?, ?)
        """, (fda_id, company_name, subject, issuing_office, posted_date, response.url, full_text))
        
        self.db.commit()
        
        # 返回数据
        yield {
            'fda_id': fda_id,
            'company_name': company_name,
            'subject': subject,
            'issuing_office': issuing_office,
            'posted_date': posted_date,
            'url': response.url,
            'full_text': full_text
        }

class FDACrawlerRunner:
    """FDA爬虫运行器"""
    
    def __init__(self):
        self.db = sqlite3.connect('/root/data/fda_warning.db')
    
    def run(self):
        """运行爬虫"""
        from scrapy.crawler import CrawlerProcess
        from scrapy.utils.project import get_project_settings
        
        process = CrawlerProcess(get_project_settings())
        process.crawl(FDALetterSpider)
        process.start()
    
    def get_latest(self, limit=10):
        """获取最新警告信"""
        cursor = self.db.execute("""
            SELECT fda_id, company_name, subject, posted_date, url
            FROM fda_warning_letters
            ORDER BY posted_date DESC
            LIMIT ?
        """, (limit,))
        
        return cursor.fetchall()
    
    def search(self, query, page=1, page_size=20):
        """搜索警告信"""
        offset = (page - 1) * page_size
        
        cursor = self.db.execute("""
            SELECT fda_id, company_name, subject, posted_date, url
            FROM fda_warning_letters
            WHERE company_name LIKE ? OR subject LIKE ? OR full_text LIKE ?
            ORDER BY posted_date DESC
            LIMIT ? OFFSET ?
        """, (f'%{query}%', f'%{query}%', f'%{query}%', page_size, offset))
        
        return cursor.fetchall()
```

```python
# scheduler.py
from apscheduler.schedulers.background import BackgroundScheduler
from datetime import datetime
import logging

class TaskScheduler:
    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.setup_logging()
    
    def setup_logging(self):
        """配置日志"""
        logging.basicConfig(
            level=logging.INFO,
            format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
            handlers=[
                logging.FileHandler('/root/logs/scheduler.log'),
                logging.StreamHandler()
            ]
        )
    
    def start(self):
        """启动调度器"""
        # FDA警告信爬虫 - 每日凌晨2点运行
        self.scheduler.add_job(
            self.run_fda_crawler,
            'cron',
            hour=2,
            minute=0,
            id='fda_crawler'
        )
        
        # 法规更新检查 - 每周一上午9点运行
        self.scheduler.add_job(
            self.check_regulation_updates,
            'cron',
            day_of_week='mon',
            hour=9,
            minute=0,
            id='regulation_checker'
        )
        
        # 行业新闻聚合 - 每天上午8点运行
        self.scheduler.add_job(
            self.aggregate_news,
            'cron',
            hour=8,
            minute=0,
            id='news_aggregator'
        )
        
        # 数据备份 - 每天凌晨3点运行
        self.scheduler.add_job(
            self.backup_data,
            'cron',
            hour=3,
            minute=0,
            id='data_backup'
        )
        
        self.scheduler.start()
        logging.info("Scheduler started")
    
    def run_fda_crawler(self):
        """运行FDA爬虫"""
        logging.info("Starting FDA crawler")
        try:
            from fda_crawler import FDACrawlerRunner
            runner = FDACrawlerRunner()
            runner.run()
            logging.info("FDA crawler completed")
        except Exception as e:
            logging.error(f"FDA crawler error: {e}")
    
    def check_regulation_updates(self):
        """检查法规更新"""
        logging.info("Checking regulation updates")
        try:
            from regulation_checker import RegulationChecker
            checker = RegulationChecker()
            updates = checker.check_all()
            logging.info(f"Found {len(updates)} regulation updates")
        except Exception as e:
            logging.error(f"Regulation checker error: {e}")
    
    def aggregate_news(self):
        """聚合行业新闻"""
        logging.info("Aggregating news")
        try:
            from news_aggregator import NewsAggregator
            aggregator = NewsAggregator()
            count = aggregator.fetch_all()
            logging.info(f"Aggregated {count} news items")
        except Exception as e:
            logging.error(f"News aggregator error: {e}")
    
    def backup_data(self):
        """备份数据"""
        logging.info("Backing up data")
        try:
            import subprocess
            from datetime import datetime
            
            date = datetime.now().strftime('%Y%m%d')
            backup_dir = f'/root/backups/fda/{date}'
            
            # 创建备份目录
            subprocess.run(['mkdir', '-p', backup_dir])
            
            # 备份数据库
            subprocess.run([
                'sqlite3', '/root/data/fda_warning.db',
                f".backup '{backup_dir}/fda_warning.db'"
            ])
            
            # 压缩备份
            subprocess.run([
                'tar', '-czf', f'{backup_dir}.tar.gz', backup_dir
            ])
            
            # 删除旧备份（保留7天）
            subprocess.run([
                'find', '/root/backups/fda', '-name', '*.tar.gz',
                '-mtime', '+7', '-delete'
            ])
            
            logging.info("Backup completed")
        except Exception as e:
            logging.error(f"Backup error: {e}")

# 启动调度器

    scheduler = TaskScheduler()
```

```python
# regulation_monitor.py
import requests
from bs4 import BeautifulSoup
import sqlite3
from datetime import datetime
import hashlib

class RegulationMonitor:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/regulations.db')
        self.create_tables()
        
        # 监控源
        self.sources = {
            'nmpa': {
                'name': '国家药品监督管理局',
                'url': 'https://www.nmpa.gov.cn/xxgk/ggtg/',
                'selector': 'li.list_item'
            },
            'cde': {
                'name': '药品审评中心',
                'url': 'https://www.cde.org.cn/main/news/listpage/545cf855a50574699b46a498aa69dd02',
                'selector': '.list-content li'
            },
            'fda': {
                'name': '美国食品药品监督管理局',
                'url': 'https://www.fda.gov/regulatory-information/search-fda-guidance-documents',
                'selector': 'article'
            }
        }
    
    def create_tables(self):
        """创建数据库表"""
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS regulations (
                id INTEGER PRIMARY KEY,
                title TEXT NOT NULL,
                source TEXT NOT NULL,
                source_url TEXT,
                content_hash TEXT UNIQUE,
                publish_date DATE,
                effective_date DATE,
                industry TEXT,
                region TEXT,
                summary TEXT,
                full_text TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS regulation_updates (
                id INTEGER PRIMARY KEY,
                regulation_id INTEGER REFERENCES regulations(id),
                update_type TEXT,
                old_value TEXT,
                new_value TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        self.db.commit()
    
    def check_source(self, source_key):
        """检查单个来源"""
        source = self.sources[source_key]
        response = requests.get(source['url'], timeout=30)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        items = soup.select(source['selector'])
        new_items = []
        
        for item in items:
            title = item.find('a').text.strip() if item.find('a') else ''
            link = item.find('a')['href'] if item.find('a') else ''
            date = item.find('span').text.strip() if item.find('span') else ''
            
            # 生成内容哈希
            content_hash = hashlib.md5(f"{title}{link}".encode()).hexdigest()
            
            # 检查是否已存在
            existing = self.db.execute(
                "SELECT id FROM regulations WHERE content_hash = ?",
                (content_hash,)
            ).fetchone()
            
            if not existing:
                # 保存新法规
                cursor = self.db.execute("""
                    INSERT INTO regulations (title, source, source_url, content_hash, publish_date, region)
                    VALUES (?, ?, ?, ?, ?, ?)
                """, (title, source['name'], link, content_hash, date, self.get_region(source_key)))
                
                new_items.append({
                    'id': cursor.lastrowid,
                    'title': title,
                    'source': source['name'],
                    'url': link,
                    'date': date
                })
        
        self.db.commit()
        return new_items
    
    def get_region(self, source_key):
        """获取地区"""
        region_map = {
            'nmpa': '中国',
            'cde': '中国',
            'fda': '美国'
        }
        return region_map.get(source_key, '未知')
    
    def check_all(self):
        """检查所有来源"""
        all_updates = []
        
        for source_key in self.sources:
            try:
                updates = self.check_source(source_key)
                all_updates.extend(updates)
                print(f"Checked {source_key}: found {len(updates)} new items")
            except Exception as e:
                print(f"Error checking {source_key}: {e}")
        
        return all_updates
    
    def get_latest(self, limit=10, region=None):
        """获取最新法规"""
        query = """
            SELECT id, title, source, publish_date, region, summary
            FROM regulations
        """
        params = []
        
        if region:
            query += " WHERE region = ?"
            params.append(region)
        
        query += " ORDER BY publish_date DESC LIMIT ?"
        params.append(limit)
        
        cursor = self.db.execute(query, params)
        return cursor.fetchall()
```

```python
# analytics.py
import sqlite3
from datetime import datetime, timedelta
import matplotlib.pyplot as plt
import io
import base64

class AnalyticsService:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/fda_warning.db')
    
    def get_monthly_trends(self, months=12):
        """获取月度趋势"""
        start_date = datetime.now() - timedelta(days=months * 30)
        
        cursor = self.db.execute("""
            SELECT 
                strftime('%Y-%m', posted_date) as month,
                COUNT(*) as count
            FROM fda_warning_letters
            WHERE posted_date >= ?
            GROUP BY month
            ORDER BY month
        """, (start_date.strftime('%Y-%m-%d'),))
        
        return cursor.fetchall()
    
    def get_office_distribution(self):
        """获取办公室分布"""
        cursor = self.db.execute("""
            SELECT 
                issuing_office,
                COUNT(*) as count
            FROM fda_warning_letters
            GROUP BY issuing_office
            ORDER BY count DESC
        """)
        
        return cursor.fetchall()
    
    def get_violation_types(self):
        """获取违规类型分布"""
        cursor = self.db.execute("""
            SELECT 
                violation_type,
                COUNT(*) as count
            FROM fda_analysis
            GROUP BY violation_type
            ORDER BY count DESC
        """)
        
        return cursor.fetchall()
    
    def get_risk_distribution(self):
        """获取风险等级分布"""
        cursor = self.db.execute("""
            SELECT 
                risk_level,
                COUNT(*) as count
            FROM fda_analysis
            GROUP BY risk_level
        """)
        
        return cursor.fetchall()
    
    def get_country_distribution(self):
        """获取国家分布"""
        cursor = self.db.execute("""
            SELECT 
                country,
                COUNT(*) as count
            FROM fda_warning_letters
            WHERE country IS NOT NULL
            GROUP BY country
            ORDER BY count DESC
            LIMIT 10
        """)
        
        return cursor.fetchall()
    
    def generate_chart(self, chart_type, data):
        """生成图表"""
        plt.figure(figsize=(10, 6))
        
        if chart_type == 'bar':
            labels = [item[0] for item in data]
            values = [item[1] for item in data]
            plt.bar(labels, values)
            plt.xticks(rotation=45, ha='right')
        
        elif chart_type == 'pie':
            labels = [item[0] for item in data]
            values = [item[1] for item in data]
            plt.pie(values, labels=labels, autopct='%1.1f%%')
        
        elif chart_type == 'line':
            labels = [item[0] for item in data]
            values = [item[1] for item in data]
            plt.plot(labels, values, marker='o')
            plt.xticks(rotation=45, ha='right')
        
        plt.tight_layout()
        
        # 保存为base64
        img_buffer = io.BytesIO()
        plt.savefig(img_buffer, format='png')
        img_buffer.seek(0)
        img_base64 = base64.b64encode(img_buffer.getvalue()).decode()
        
        plt.close()
        
        return img_base64
    
    def get_dashboard_data(self):
        """获取仪表盘数据"""
        return {
            'total_letters': self.db.execute("SELECT COUNT(*) FROM fda_warning_letters").fetchone()[0],
            'active_letters': self.db.execute("SELECT COUNT(*) FROM fda_warning_letters WHERE status = 'active'").fetchone()[0],
            'monthly_trends': self.get_monthly_trends(),
            'office_distribution': self.get_office_distribution(),
            'violation_types': self.get_violation_types(),
            'risk_distribution': self.get_risk_distribution(),
            'country_distribution': self.get_country_distribution()
        }
```
