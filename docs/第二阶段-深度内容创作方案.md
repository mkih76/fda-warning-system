## 三、搜索功能优化方案

### 3.1 全文搜索实现

```python
# search_engine.py
import sqlite3
from whoosh.index import create_in
from whoosh.fields import Schema, TEXT, ID, DATETIME
from whoosh.qparser import MultifieldParser
import os

class SearchEngine:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/fda_warning.db')
        self.index_dir = '/root/data/search_index'
        
        # 创建索引目录
        if not os.path.exists(self.index_dir):
            os.makedirs(self.index_dir)
        
        # 定义搜索模式
        self.schema = Schema(
            id=ID(stored=True, unique=True),
            title=TEXT(stored=True),
            content=TEXT(stored=True),
            category=TEXT(stored=True),
            tags=TEXT(stored=True),
            date=DATETIME(stored=True)
        )
        
        # 创建索引
        self.ix = create_in(self.index_dir, self.schema)
    
    def index_warning_letters(self):
        """索引FDA警告信"""
        writer = self.ix.writer()
        
        # 查询警告信
        letters = self.db.execute("""
            SELECT l.id, l.company_name, l.subject, l.full_text,
                   a.translation_zh, a.summary_zh, l.posted_date
            FROM fda_warning_letters l
            LEFT JOIN fda_analysis a ON l.id = a.letter_id
        """).fetchall()
        
        for letter in letters:
            writer.add_document(
                id=str(letter[0]),
                title=letter[2],
                content=f"{letter[3] or ''} {letter[4] or ''} {letter[5] or ''}",
                category="FDA警告信",
                tags=letter[1],
                date=letter[6]
            )
        
        writer.commit()
    
    def index_regulations(self):
        """索引法规信息"""
        writer = self.ix.writer()
        
        # 查询法规
        regulations = self.db.execute("""
            SELECT id, title, summary, full_text, region, publish_date
            FROM regulations
        """).fetchall()
        
        for reg in regulations:
            writer.add_document(
                id=f"reg_{reg[0]}",
                title=reg[1],
                content=f"{reg[2] or ''} {reg[3] or ''}",
                category="法规信息",
                tags=reg[4],
                date=reg[5]
            )
        
        writer.commit()
    
    def search(self, query, category=None, page=1, page_size=20):
        """执行搜索"""
        with self.ix.searcher() as searcher:
            # 构建查询
            if category:
                query_str = f"category:{category} AND {query}"
            else:
                query_str = query
            
            parser = MultifieldParser(["title", "content", "tags"], self.ix.schema)
            q = parser.parse(query_str)
            
            # 执行搜索
            results = searcher.search(q, limit=page * page_size)
            
            # 分页
            start = (page - 1) * page_size
            end = start + page_size
            page_results = results[start:end]
            
            return {
                "total": len(results),
                "items": [dict(hit) for hit in page_results],
                "page": page,
                "page_size": page_size
            }

# 初始化搜索引擎
search_engine = SearchEngine()
search_engine.index_warning_letters()
search_engine.index_regulations()
```

### 3.2 搜索API

```python
# search_api.py
from fastapi import FastAPI, Query
from typing import Optional

app = FastAPI()

@app.get("/api/search")
async def search(
    q: str = Query(..., min_length=1),
    category: Optional[str] = None,
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100)
):
    """搜索API"""
    from search_engine import search_engine
    
    results = search_engine.search(q, category, page, page_size)
    
    return {
        "query": q,
        "category": category,
        "total": results["total"],
        "items": results["items"],
        "page": results["page"],
        "page_size": results["page_size"]
    }

@app.get("/api/search/suggest")
async def search_suggest(q: str = Query(..., min_length=2)):
    """搜索建议"""
    from search_engine import search_engine
    
    # 获取搜索建议
    suggestions = search_engine.get_suggestions(q)
    
    return {
        "query": q,
        "suggestions": suggestions
    }
```

### 4.2 用户API

```python
# user_api.py
from fastapi import FastAPI, Depends, HTTPException
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from pydantic import BaseModel
import hashlib
import jwt
from datetime import datetime, timedelta

app = FastAPI()
security = HTTPBearer()

# 配置
SECRET_KEY = "your-secret-key"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30

class UserCreate(BaseModel):
    username: str
    email: str
    password: str
    company: str = None
    industry: str = None

class UserLogin(BaseModel):
    username: str
    password: str

def hash_password(password: str) -> str:
    """密码哈希"""
    return hashlib.sha256(password.encode()).hexdigest()

def create_access_token(data: dict):
    """创建访问令牌"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

@app.post("/api/auth/register")
async def register(user: UserCreate):
    """用户注册"""
    from main import db
    
    # 检查用户名是否已存在
    existing = db.execute(
        "SELECT id FROM users WHERE username = ? OR email = ?",
        (user.username, user.email)
    ).fetchone()
    
    if existing:
        raise HTTPException(status_code=400, detail="用户名或邮箱已存在")
    
    # 创建用户
    password_hash = hash_password(user.password)
    db.execute(
        """INSERT INTO users (username, email, password_hash, company, industry)
           VALUES (?, ?, ?, ?, ?)""",
        (user.username, user.email, password_hash, user.company, user.industry)
    )
    db.commit()
    
    return {"message": "注册成功"}

@app.post("/api/auth/login")
async def login(user: UserLogin):
    """用户登录"""
    from main import db
    
    # 验证用户
    password_hash = hash_password(user.password)
    user_data = db.execute(
        "SELECT id, username, email FROM users WHERE username = ? AND password_hash = ?",
        (user.username, password_hash)
    ).fetchone()
    
    if not user_data:
        raise HTTPException(status_code=401, detail="用户名或密码错误")
    
    # 创建令牌
    access_token = create_access_token(data={"sub": user_data[1]})
    
    # 更新最后登录时间
    db.execute(
        "UPDATE users SET last_login = CURRENT_TIMESTAMP WHERE id = ?",
        (user_data[0],)
    )
    db.commit()
    
    return {
        "access_token": access_token,
        "token_type": "bearer",
        "user": {
            "id": user_data[0],
            "username": user_data[1],
            "email": user_data[2]
        }
    }

@app.get("/api/user/profile")
async def get_profile(credentials: HTTPAuthorizationCredentials = Depends(security)):
    """获取用户信息"""
    from main import db
    
    try:
        payload = jwt.decode(credentials.credentials, SECRET_KEY, algorithms=[ALGORITHM])
        username = payload.get("sub")
    except:
        raise HTTPException(status_code=401, detail="无效的令牌")
    
    user = db.execute(
        "SELECT id, username, email, company, industry FROM users WHERE username = ?",
        (username,)
    ).fetchone()
    
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    return {
        "id": user[0],
        "username": user[1],
        "email": user[2],
        "company": user[3],
        "industry": user[4]
    }

@app.post("/api/user/favorites")
async def add_favorite(
    item_type: str,
    item_id: int,
    credentials: HTTPAuthorizationCredentials = Depends(security)
):
    """添加收藏"""
    from main import db
    
    try:
        payload = jwt.decode(credentials.credentials, SECRET_KEY, algorithms=[ALGORITHM])
        username = payload.get("sub")
    except:
        raise HTTPException(status_code=401, detail="无效的令牌")
    
    user = db.execute("SELECT id FROM users WHERE username = ?", (username,)).fetchone()
    
    # 添加收藏
    db.execute(
        """INSERT OR IGNORE INTO user_favorites (user_id, item_type, item_id)
           VALUES (?, ?, ?)""",
        (user[0], item_type, item_id)
    )
    db.commit()
    
    return {"message": "添加成功"}

@app.get("/api/user/favorites")
async def get_favorites(credentials: HTTPAuthorizationCredentials = Depends(security)):
    """获取收藏列表"""
    from main import db
    
    try:
        payload = jwt.decode(credentials.credentials, SECRET_KEY, algorithms=[ALGORITHM])
        username = payload.get("sub")
    except:
        raise HTTPException(status_code=401, detail="无效的令牌")
    
    user = db.execute("SELECT id FROM users WHERE username = ?", (username,)).fetchone()
    
    favorites = db.execute(
        """SELECT item_type, item_id, created_at 
           FROM user_favorites 
           WHERE user_id = ?
           ORDER BY created_at DESC""",
        (user[0],)
    ).fetchall()
    
    return {
        "favorites": [
            {
                "type": fav[0],
                "id": fav[1],
                "created_at": fav[2]
            }
            for fav in favorites
        ]
    }
```
