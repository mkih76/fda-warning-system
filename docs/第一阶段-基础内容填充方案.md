-- FDA警告信主表
CREATE TABLE fda_warning_letters (
    id INTEGER PRIMARY KEY,
    fda_id TEXT UNIQUE NOT NULL,          -- FDA唯一标识
    company_name TEXT NOT NULL,            -- 公司名称
    subject TEXT,                          -- 主题
    issuing_office TEXT,                   -- 签发办公室
    issue_date DATE,                       -- 签发日期
    posted_date DATE,                      -- 公布日期
    closeout_date DATE,                    -- 关闭日期
    status TEXT DEFAULT 'active',          -- 状态
    country TEXT,                          -- 国家
    region TEXT,                           -- 地区
    fei_number TEXT,                       -- FEI编号
    url TEXT,                              -- FDA原始链接
    full_text TEXT,                        -- 英文全文
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- AI分析表
CREATE TABLE fda_analysis (
    id INTEGER PRIMARY KEY,
    letter_id INTEGER REFERENCES fda_warning_letters(id),
    translation_zh TEXT,                   -- 中文翻译
    summary_zh TEXT,                       -- 中文摘要
    summary_en TEXT,                       -- 英文摘要
    key_findings JSON,                     -- 关键发现
    risk_level TEXT,                       -- 风险等级
    violation_type TEXT,                   -- 违规类型
    model_used TEXT,                       -- 使用的AI模型
    confidence_score REAL,                 -- 置信度
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- CFR引用表
CREATE TABLE fda_citations (
    id INTEGER PRIMARY KEY,
    letter_id INTEGER REFERENCES fda_warning_letters(id),
    cfr_section TEXT NOT NULL,             -- CFR条款号
    cfr_title TEXT,                        -- CFR标题
    cfr_text TEXT,                         -- CFR内容
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 违规分类表
CREATE TABLE fda_violations (
    id INTEGER PRIMARY KEY,
    letter_id INTEGER REFERENCES fda_warning_letters(id),
    system_category TEXT,                  -- 系统分类
    violation_type TEXT,                   -- 违规类型
    severity TEXT,                         -- 严重程度
    description TEXT,                      -- 描述
    description_zh TEXT,                   -- 中文描述
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 企业信息表
CREATE TABLE companies (
    id INTEGER PRIMARY KEY,
    name TEXT UNIQUE NOT NULL,             -- 公司名称
    country TEXT,                          -- 国家
    industry TEXT,                         -- 行业
    warning_count INTEGER DEFAULT 0,       -- 警告信数量
    risk_level TEXT,                       -- 风险等级
    last_warning_date DATE,                -- 最近警告日期
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

### 2.2 数据备份策略

```bash
# 每日自动备份
#!/bin/bash
BACKUP_DIR="/root/backups/fda"
DATE=$(date +%Y%m%d)
DB_PATH="/root/data/fda_warning.db"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 备份数据库
sqlite3 $DB_PATH ".backup '$BACKUP_DIR/fda_$DATE.db'"

# 压缩备份
tar -czf "$BACKUP_DIR/fda_$DATE.tar.gz" "$BACKUP_DIR/fda_$DATE.db"

# 删除7天前的备份
find $BACKUP_DIR -name "*.tar.gz" -mtime +7 -delete

echo "Backup completed: fda_$DATE.tar.gz"
```

def export_to_csv():
    """导出为CSV格式"""
    import csv
    import io
    
    output = io.StringIO()
    writer = csv.writer(output)
    
    # 写入标题
    writer.writerow([
        "FDA ID", "公司名称", "主题", "签发办公室",
        "签发日期", "状态", "风险等级", "违规类型"
    ])
    
    # 查询数据

        SELECT l.fda_id, l.company_name, l.subject, l.issuing_office,
               l.issue_date, l.status, a.risk_level, a.violation_type
        FROM fda_warning_letters l
        LEFT JOIN fda_analysis a ON l.id = a.letter_id
    """).fetchall()
    
    # 写入数据
    for letter in letters:
        writer.writerow(letter)

### 3.2 数据采集脚本

```python
# regulations_crawler.py
import requests
from bs4 import BeautifulSoup
import sqlite3
from datetime import datetime

class RegulationCrawler:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/regulations.db')
        self.create_tables()
    
    def create_tables(self):
        """创建法规信息表"""
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS regulations (
                id INTEGER PRIMARY KEY,
                title TEXT NOT NULL,
                source TEXT NOT NULL,
                source_url TEXT,
                publish_date DATE,
                effective_date DATE,
                industry TEXT,
                region TEXT,
                summary TEXT,
                full_text TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.db.commit()
    
    def crawl_nmpa(self):
        """爬取NMPA法规"""
        url = "https://www.nmpa.gov.cn/xxgk/ggtg/"
        response = requests.get(url)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # 解析法规列表
        items = soup.find_all('li', class_='list_item')
        for item in items:
            title = item.find('a').text.strip()
            link = item.find('a')['href']
            date = item.find('span').text.strip()
            
            # 保存到数据库
            self.db.execute("""
                INSERT INTO regulations (title, source, source_url, publish_date, region)
                VALUES (?, ?, ?, ?, ?)
            """, (title, 'NMPA', link, date, '中国'))
        
        self.db.commit()
    
    def crawl_fda_regulations(self):
        """爬取FDA法规"""
        url = "https://www.fda.gov/regulatory-information/search-fda-guidance-documents"
        response = requests.get(url)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # 解析法规列表
        items = soup.find_all('article')
        for item in items:
            title = item.find('h3').text.strip()
            link = item.find('a')['href']
            date = item.find('time').text.strip()
            
            # 保存到数据库
            self.db.execute("""
                INSERT INTO regulations (title, source, source_url, publish_date, region)
                VALUES (?, ?, ?, ?, ?)
            """, (title, 'FDA', link, date, '美国'))
        
        self.db.commit()

# 运行爬虫
crawler = RegulationCrawler()
crawler.crawl_nmpa()
crawler.crawl_fda_regulations()
```

```python
# news_aggregator.py
import feedparser
import sqlite3
from datetime import datetime

class NewsAggregator:
    def __init__(self):
        self.db = sqlite3.connect('/root/data/news.db')
        self.create_tables()
        
        # RSS订阅源
        self.feeds = {
            'fda_news': 'https://www.fda.gov/about-fda/contact-fda/stay-informed/rss-feeds/press-releases/rss.xml',
            'nmpa_news': 'https://www.nmpa.gov.cn/rss/news.xml',
            'pharma_news': 'https://www.pharmaceutical-technology.com/feed/',
        }
    
    def create_tables(self):
        """创建资讯表"""
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS news (
                id INTEGER PRIMARY KEY,
                title TEXT NOT NULL,
                source TEXT NOT NULL,
                source_url TEXT,
                publish_date DATE,
                category TEXT,
                tags TEXT,
                summary TEXT,
                content TEXT,
                is_read BOOLEAN DEFAULT FALSE,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.db.commit()
    
    def fetch_rss(self, feed_name):
        """获取RSS源"""
        feed = feedparser.parse(self.feeds[feed_name])
        
        for entry in feed.entries:
            title = entry.title
            link = entry.link
            published = entry.published
            summary = entry.summary
            
            # 保存到数据库
            self.db.execute("""
                INSERT OR IGNORE INTO news (title, source, source_url, publish_date, summary)
                VALUES (?, ?, ?, ?, ?)
            """, (title, feed_name, link, published, summary))
        
        self.db.commit()
    
    def fetch_all(self):
        """获取所有订阅源"""
        for feed_name in self.feeds:
            self.fetch_rss(feed_name)
            print(f"Fetched {feed_name}")

# 运行聚合器
aggregator = NewsAggregator()
aggregator.fetch_all()
```
